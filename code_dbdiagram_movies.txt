Table movie{
  movie_id int [primary key, increment]
  title varchar(1000) 
  budget int 
  homepage varchar(1000) 
  overview varchar(1000)
  popularity decimal(12,6)
  release_date date
  revenue bigint
  runtime int
  movie_status varchar(50)
  tagline varchar(1000)
  vote_average decimal(4,2)
  vote_count int
}

table country{
  country_id int [pk, increment]
  country_iso_code varchar(10)
  country_name varchar(200)
}

table department{
  department_id int [pk, increment]
  department_name varchar(200)
}

table gender{
  gender_id int [pk]
  gender varchar(20)
}

table genre{
  genre_id int [pk]
  genre_name varchar(100)
}

table keyword{
  keyword_id int [pk]
  keyword_name varchar(100)
}

table language{
  language_id int [pk, increment]
  language_code varchar(10)
  language_name varchar(500)
}

table language_role{
  role_id int [pk]
  language_role varchar(20)
}

table movie_cast{
  movie_id int
  person_id int
  character_name varchar(400)
  gender_id int
  cast_order int
}

table movie_company{
  movie_id int
  company_id int
}

table movie_crew{
  movie_id int
  person_id int
  department_id int
  job varchar(200)
}

table movie_genres{
  movie_id int
  genre_id int
}

table movie_keywords{
  movie_id int
  keyword_id int 
}

table movie_language{
  movie_id int
  languague_id int
  language_role_id int
}

table person{
  person_id int [pk]
  person_name varchar(500)
}

table production_company{
  company_id int
  company_name varchar(200)
}

table production_country{
  movie_id int
  country_id int
}
